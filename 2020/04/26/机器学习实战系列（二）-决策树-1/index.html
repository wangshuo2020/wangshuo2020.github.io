def createTree(dataSet,labels): //返回一个树的字典
classList = [example[-1] for example in dataSet] //从数据集中取出类别列表
if classList.count(classList[0]) == len(classList): //类别列表中只有类别classList[0]
return classList[0] #stop splitting when all of the classes are equal
if len(dataSet[0]) == 1: #stop splitting when there are no more features in dataSet //dataSet中元素的长度为1
return majorityCnt(classList) //上面两个条件就是数据集递归的终止条件，因为每分类一次数据集中元素长度就会减1
bestFeat = chooseBestFeatureToSplit(dataSet)
bestFeatLabel = labels[bestFeat] //找出分类的最好的特征
myTree = {bestFeatLabel:{}} //建立一个空值的字典
del(labels[bestFeat]) //删掉分类最好的那个特征
featValues = [example[bestFeat] for example in dataSet] //取出所有最好特征的值
uniqueVals = set(featValues) //去掉重复的元素
for value in uniqueVals:
subLabels = labels[:] #copy all of labels, so trees don’t mess up existing labels
myTree[bestFeatLabel][value] = createTree(splitDataSet(dataSet, bestFeat, value),subLabels) //在子数据集和子标签列表上递归调用createTree函数
return myTree //嵌套的字典怎么索引？字典的值可以是一个字典。myTree[bestFeatLabel]是一个字典。索引嵌套字典值的方法与索引一个二维数组的元素相同。
